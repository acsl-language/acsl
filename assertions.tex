\begin{syntax}
  C-compound-statement ::= "{" C-declaration* C-statement* assertion+ "}"
        \
  C-statement ::= assertion C-statement \
  assertion-kind ::= "assert" | clause-kind \
  assertion ::= "/*@" assertion-kind pred ";" "*/" ;
  | "/*@" "for" id ("," id)* ":" ;
      assertion-kind pred ";" "*/" ;
\end{syntax}
