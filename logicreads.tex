  logic-function-decl ::= "logic" type-expr poly-id;
  parameters reads-clause ";"
  \
  logic-predicate-decl ::= "predicate" poly-id ;
  parameters? reads-clause ";"
  \
  reads-clause ::= "reads" locations
  \

                         parameters {reads-clause} "=" term ";"

  parameters? {reads-clause} "=" pred ";"
\end{syntax}
